/entry (NXentry)
  definition = "NXsas_raw"
  schema_version = "1.0"                      (recommandé)
  title, start_time, end_time, duration       (recommandé)

  /sample (NXsample)
    name
    thickness [m]                             (recommandé)
    transmission [1]                          (optionnel brut)
    (environment logs optionnels)

  /instrument (NXinstrument)
    /geometry (NXcollection)                  # SCARLET reserved
      origin_definition = "sample center"
      axis_convention   = "+z downstream, +x beam-right, +y up"
      units_length = "m"
      units_angle  = "deg"

    /source (NXsource)
      name, type, probe

    /monochromator (NXmonochromator)
      wavelength [angstrom]                   (mono)
      wavelength_error [angstrom]             # écart-type (ou wavelength_errors)
      /velocity_selector (NXvelocity_selector) (optionnel)
        rotation_speed [rpm] (optionnel)
        selector_type, tilt, etc. (optionnel)

    /collimation (NXcollection)               # ta “bonne” collimation détaillée
      (voir section 2 plus bas)

    /beamstop (NXbeamstop)
      shape = "circular"|"rectangular"|...
      diameter [m] / x_gap,y_gap [m]
      /transformations (NXtransformations)
        translation [m] = (x,y,z)

    /detector0 (NXdetector)
      detector_number = 0
      x_pixel_size [m]
      y_pixel_size [m]
      x_pixels, y_pixels
      /transformations (NXtransformations)
        translation [m] = (x,y,z)             # z>0 aval
        rotation [deg] (optionnel)
      counts (dataset)                        # où vivent réellement les données
      counts_errors (dataset, optionnel)
      pixel_mask_static (optionnel)
      sensitivity_map (optionnel)

    /detector1 (NXdetector)
      ... idem ...
    /detectorN (NXdetector)
      ...

    /monitor0 (NXmonitor)
        mode    = "monitor"
        preset  = 5.0e7
        integral= 4.83e7
        count_time = 600.0
        /transformations (NXtransformations)   # optionnel
            translation [m] = (x,y,z)
            rotation    [deg] (optionnel)



    /monitor1 (NXmonitor)
      ...

/entry/control (NXmonitor)
  mode    = "timer"
  preset  = 600.0
  count_time = 600.2

/entry/data0 (NXdata)         # NXlink vers detector0/counts (+ axes)
    @signal = "counts"
    @axes = ["y","x"] ou ["wavelength","y","x"]
    counts  -> /entry/instrument/detector0/counts
    counts_errors -> /entry/instrument/detector0/counts_errors (si existe)
    x, y (optionnel, indices pixel) ou juste axes implicites

/entry/data1 (NXdata)
    counts -> /entry/instrument/detector1/counts
    ...
/entry/dataN (NXdata)
    ...


/entry/instrument/collimation (NXcollection)
  @description = "SCARLET collimation chain"
  element_order = ["A0","S1","G2","A3"]

  /elements (NXcollection)
    /A0 (NXaperture)
      shape, x_gap/y_gap ou diameter
      /transformations
        translation [m] = (x,y,z)   # z<0 amont
    /S1 (NXslit)
      x_gap, y_gap, x_center, y_center
      /transformations/translation
    /G2 (NXguide)
      length, entrance/exit sizes, m_value, curvature_radius (opt.)
      /transformations/translation  # convention: centre du plan d’entrée
    /A3 (NXaperture)
      ...
